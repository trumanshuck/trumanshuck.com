import { Box, Text } from 'rebass'

<Box mb={4} sx={{whiteSpace: 'nowrap'}}>
  <Text>We slept in the closet</Text>
  <Text>on a queenspring;</Text>
  <Text>white walls sheltered cloud-</Text>
  <Text>stitched blankets and</Text>
  <Text>pillows with the shape of each of</Text>
  <Text>Christopher Robin’s friends.</Text>
  <Text>A map of the world crowded</Text>
  <Text>rest of the space; there was room</Text>
  <Text>enough for our bed</Text>
  <Text mb={3}>but none for the boogeyman.</Text>

  <Text>Outside lay gleamingblocks;</Text>
  <Text>our traps hidden in thick carpet</Text>
  <Text>and a huge wooden castle dad</Text>
  <Text>had built. It’s walls higher than</Text>
  <Text>tiptoes and strong enough to</Text>
  <Text mb={3}>keep even the dark things out.</Text>

  <Text>And when the news came on</Text>
  <Text>or soft sleepyeyes settled over the house</Text>
  <Text>mom lead us back upstairs;</Text>
  <Text>our small feet scratchingmice on hardwood</Text>
  <Text>and we would shriek into bed</Text>
  <Text>using the last of our energy</Text>
  <Text mb={3}>in jumbling the sheets.</Text>

  <Text>She would smile</Text>
  <Text>foreheadkisses dream well</Text>
  <Text>but I preferred the grasp</Text>
  <Text>of cool, solid earlobe;</Text>
  <Text>It felt safe</Text>
  <Text mb={3}>and fit in my tiny hands.</Text>

  <Text>And when she left,</Text>
  <Text>though the moonlight threw shadows</Text>
  <Text>that hung watching through</Text>
  <Text>the door sliveredopen,</Text>
  <Text>my hand held the impression of</Text>
  <Text>a small pink shield and so</Text>
  <Text>the silvered light</Text>
  <Text>bounced straight from block</Text>
  <Text>to castle to dreams.</Text>
</Box>
