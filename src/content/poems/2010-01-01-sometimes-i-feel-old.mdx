import { Text } from 'rebass';

<Text>Sometimes I feel old; I can see</Text>
<Text>time collecting:</Text>
<Text pl={3}>a fat water droplet bowing some</Text>
<Text pl={4}>thick grass stem</Text>
<Text pl={4}>its weight an imperfect circle back to the earth</Text>
<Text pl={3}>holding on impossibly lo</Text>
<Text pl={3}>ng before rolling regretfully off.</Text>
<Text>years are splashed so roughly</Text>
<Text mb={3}>against concrete</Text>

<Text>and memory is spread so clearly,</Text>
<Text>one fragile film absorbed with a glance</Text>
<Text pl={3}><in></in>but the water is thin against the pavement</Text>
<Text pl={3}><in></in>and black dust floats near the top</Text>
<Text pl={4}><in></in><in></in> obscuring the speckles and liquid beneath</Text>
<Text>now each old rainbow bounce</Text>
<Text mb={3}>suddenly lost</Text>

<Text>but another drop is building; anoth</Text>
<Text>er drop will fall as dew</Text>
<Text>condenses another</Text>
<Text>drop</Text>
