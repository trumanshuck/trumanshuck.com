import { Text } from 'rebass'

<Text>I walk a little more slowly toward the horizon</Text>
<Text>than I did before</Text>
<Text pl={3}>(each step a question</Text>
<Text pl={3} mb={3}>answered by an echo)</Text>

<Text>I know the sea, still, but hear the waves'</Text>
<Text>crash fuzzyfar away</Text>
<Text pl={3}>(a bone china plate dropped</Text>
<Text pl={3} mb={3}>on the other side of the house)</Text>

<Text>And one leg drags the other, now, and my stride</Text>
<Text>leaves a small trough in the sand</Text>
<Text>And I see the world narrowed by high</Text>
<Text mb={3}>gray cliffs.</Text>

<Text>I remember leaving distinct craters, every forward</Text>
<Text>push a burst of earth under a light more gold</Text>
<Text mb={3}>than red</Text>

<Text>The earth tumbling along before me</Text>
<Text>The air ahead unoccupied</Text>
<Text mb={3}>Awaiting.</Text>

<Text>I can see each breath, now, and I wonder which</Text>
<Text mb={3}>misted ghost I've exhaled</Text>

<Text>But the surf still whispers endlessly ahead</Text>
<Text>A gull sings sharply</Text>
<Text>Beckoning.</Text>
