import { Box, Text } from 'rebass'

<Box sx={{whiteSpace: 'nowrap'}}>
  <Text>When I die:</Text>
  <Text pl={3}>with a skin papyrus paper</Text>
  <Text pl={3}>and shakenbreath trembling</Text>
  <Text pl={3}>eyes like ocean fog</Text>
  <Text mb={3}>well,</Text>

  <Text>When I die I</Text>
  <Text pl={3}>doubt I’ll swim forever</Text>
  <Text pl={3}>in the highest clouds</Text>
  <Text pl={3}>propped up with an alabaster spyglass</Text>
  <Text pl={3}>and peeping grin of the curious</Text>
  <Text mb={3}>no.</Text>

  <Text>When I die I think</Text>
  <Text pl={3}>I’ll mix with the sea as ash</Text>
  <Text pl={3}>and get swallowed by a tiny silver fish to</Text>
  <Text pl={3}>ply the ocean’s undercurrent</Text>
  <Text pl={3}>my soul a small passenger</Text>
  <Text pl={3}>in one creature’s belly or</Text>
  <Text mb={3}>another</Text>

  <Text>And once I’ve learned all there is</Text>
  <Text pl={3}>about coral and tides and</Text>
  <Text pl={3}>the words to each whale’s song</Text>
  <Text>perhaps then I’ll flop back onto the sand and</Text>
  <Text>stretch my new muscles and</Text>
  <Text>wave again to the sun.</Text>
</Box>
