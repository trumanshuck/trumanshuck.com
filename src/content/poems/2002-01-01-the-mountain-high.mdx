import { Box, Text } from 'rebass'

<Box mb={4} sx={{whiteSpace: 'nowrap'}}>
  <Text>The mountain high</Text>
  <Text>the valley low</Text>
  <Text>the ducks go quack!</Text>
  <Text mb={3}>moo! Cows go.</Text>

  <Text>The river swift</Text>
  <Text>the ocean deep</Text>
  <Text>the days go by</Text>
  <Text mb={3}>look! A sheep!</Text>

  <Text>Everyone turns</Text>
  <Text>and they stare</Text>
  <Text>look, they're looking</Text>
  <Text mb={3}>at the hare!</Text>

  <Text>The animals</Text>
  <Text>at work and play</Text>
  <Text>are so fuzzy</Text>
  <Text mb={3}>aren't they?</Text>

  <Text>The lakes and ponds</Text>
  <Text>serene and calm</Text>
  <Text>the trees stand tall</Text>
  <Text mb={3}>from oak to palm.</Text>

  <Text>The grass is lush</Text>
  <Text>like the softest bed</Text>
  <Text>nothing at all</Text>
  <Text mb={3}>to fear, to dread.</Text>

  <Text>In a clearing</Text>
  <Text>a summer home</Text>
  <Text>lies waiting for all</Text>
  <Text mb={3}>who wander or roam.</Text>
</Box>
