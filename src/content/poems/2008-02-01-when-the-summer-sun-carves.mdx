import { Box, Text } from 'rebass'

<Box sx={{whiteSpace: 'nowrap'}}>
  <Text>When the summer sun carves</Text>
  <Text>a boiling trough across my back,</Text>
  <Text>I think of reaching through heavy air</Text>
  <Text>and grasping a silver handle,</Text>
  <Text>letting water drawn</Text>
  <Text>from the deeplake grow</Text>
  <Text mb={3}>cool across my index finger.</Text>

  <Text>It seems, though, that it</Text>
  <Text>will never quite chill enough</Text>
  <Text>to fill completely whatever hole</Text>
  <Text mb={3}>the concentrated sunlight bore through my flesh</Text>

  <Text>even if it were to freeze down</Text>
  <Text>the spout, cracking arthriticfingers</Text>
  <Text>gripping my hand in ice</Text>
  <Text pl={3}>and even if that ice were</Text>
  <Text pl={3}>the deepest lumbering glacier</Text>
  <Text pl={3}>and I could suck each melting drop</Text>
  <Text pl={3} mb={3}>directly from it</Text>

  <Text>And just that attempt would cause my teeth</Text>
  <Text>to chatter and crack, a soft</Text>
  <Text>sprinkling rain of bonemeal.</Text>
</Box>
